# Two object linking
```
let obj = {
  name: "vivek",
  sayHello: function () {
    console.log("Hello");
  },
};

let obj2 = {
  salary: 100000,
  work: function () {
    console.log("Working");
  },
};

Object.setPrototypeOf(obj2, obj);

obj2.sayHello();

console.log(obj2);

```
